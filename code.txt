	private void checking() {	
		for (int i = 0; i < 6; i++) {
			lottoarray[i] = Integer.parseInt(balls[i]);
	            for(int g=0; g< 6;g++) {
	            	carray[g] = Integer.parseInt(choicenums[g]);
		            if (carray[g] == lottoarray[i]) {           
		               count++;
		            }
	            }
	        }
	        if (count==3) {
	        	makeToast("You Got 3 Numbers");
	        	dmon = dmon + (100 * multiplier);
	        	bmon.setText("" + dmon);
	        	reset();
	        } else if (count==4) {
	        	makeToast("You Got 4 Numbers");
	        	dmon = dmon + (1500 * multiplier);
	        	bmon.setText("" + dmon);
	        	reset();
	        } else if (count==5) {
	        	makeToast("You Got 5 Numbers");
	        	dmon = dmon + (50000 * multiplier);
	        	bmon.setText("" + dmon);
	        	reset();
	        } else if (count==6) {
	        	makeToast("You Got 6 Numbers");
	        	dmon = dmon + 9000000;
	        	bmon.setText("" + dmon);
	        	reset();
	        } else {
	        	makeToast("You Lose");
	        	bmon.setText("" + dmon);
	        	reset();
	        }
	}
	
	private void zeroBalance() {
			if ( dmon == 0 || dmon < 0 ) {
				genBut.setEnabled (false); 
				resBut.setEnabled(false);
				makeToast("Insufficient Balance, Thank You for Playing!");	
			} else {
				reset();
				sel = false;
				click = false;
				choiceButtons();
				genBut.setEnabled(true); 
				resBut.setEnabled(false);
			}
	}

	private void reset() {
		for (int i = 0; i < 6; i++) {
			lottoarray[i]= 0;
            	for(int g=0; g< 6;g++) {
            		sel = false;
            		click = false;
            		carray[g] = 0;
            		count=0;
            	}
        }
	}
